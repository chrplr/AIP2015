def map2(function, arguments):
    result_list = []
    for arg in arguments:
        result_list.append(function(arg))
    return result_list

def square(n):
    return n * n
  1. What will be the result of map2(square, [1, 2, 3, 4, 5])

def strange_loop(l):
    out = []

        out.append(item)

    return out
  1. What will be the result of strange_loop([1, 2, 5, 9, None, 3, 1, 1, None])
